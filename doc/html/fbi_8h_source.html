01016 
01051   static void scan(
01052     std::vector<const key_type *> & pointsPtrVector, //Points
01053     std::vector<const key_type *> & intervalsPtrVector,  //Intervals
01054     const typename std::tuple_element<Dim, key_type>::type::first_type & lowerBound,
01055     const typename std::tuple_element<Dim, key_type>::type::first_type & upperBound,
01056     State & state,
01057     ResultType & resultVector
01058     ) {
01059 
01060     typedef typename std::tuple_element<Dim, key_type>::type Key;
01061     typedef typename std::tuple_element<Dim, comp_type>::type Comp;
01062     Comp less;
01063 
01064     if (
01065       pointsPtrVector.size() == 0 || 
01066       intervalsPtrVector.size() == 0 || 
01067       !less(lowerBound, upperBound) 
01068     ) {
01069       return;
01070     }
01071     // switch into scanning mode if set sizes fall under the threshold
01072     if (
01073       pointsPtrVector.size() < state.getCutoff() || 
01074       intervalsPtrVector.size() < state.getCutoff() 
01075     ) {
01076       sortContainerHead<Dim>(pointsPtrVector);
01077       sortContainerHead<Dim>(intervalsPtrVector);
01078       OneWayScanner<PointsContainQueries, Dim>::
01079         scan(pointsPtrVector, intervalsPtrVector, state, resultVector);
01080       return;
01081     }
01082     // Set sizes are still above the threshold. We follow a divide and conquer
01083     // scheme: determine the median in the current dimension and use the
01084     // value to split the intervals and points for the next (recursive) call.
01085     size_t heuristicHeight = state.heuristicHeight(intervalsPtrVector.size());
01086     // FIXME: explain why you draw from the interval vector. This is the major
01087     // contribution of the implementation!!!
01088     typename Key::first_type median = 
01089       getApproxMedian<Dim>(intervalsPtrVector, heuristicHeight, state, less);
01090     
01091     std::vector<const key_type *> 
01092       intervalsMiddle, intervalsLeft, intervalsRight;
01093 
01094     typename std::vector<const key_type *>::const_iterator intVectorIt = 
01095       intervalsPtrVector.begin();
01096 
01097     while (intVectorIt != intervalsPtrVector.end()){
01098       Key interval = getKey<Dim>(*intVectorIt);
01099       const key_type * intPtr = *intVectorIt;
